    /// <summary>
    /// Example: Manually validate a sequence using spec.Allows().
    /// Useful for hand-written scenarios or debugging.
    /// </summary>
    [Test]
    public async Task ManualSequence_UsingAllows()
    {
        var spec = {{Name}}Spec.Create();
        var stateProfile = new StateProfile(new {{Name}}State());

        // Step 1: Create an entity
        var entity = new {{Name}}Entity { Id = "manual1", Name = "Manual Test" };
        var createResponse = await CreateAsync(_httpClient, entity);
        
        (bool isValid, string message, stateProfile) = spec.Allows(
            spec.GetOperation<{{Name}}Entity, ApiResult<{{Name}}Entity>>("Create"),
            entity,
            createResponse,
            stateProfile);
        Assert.That(isValid, Is.True, message);

        // Step 2: Get the entity
        var getResponse = await GetAsync(_httpClient, "manual1");
        
        (isValid, message, stateProfile) = spec.Allows(
            spec.GetOperation<string, ApiResult<{{Name}}Entity>>("Get"),
            "manual1",
            getResponse,
            stateProfile);
        Assert.That(isValid, Is.True, message);

        // Step 3: Delete the entity
        var deleteResponse = await DeleteAsync(_httpClient, "manual1");
        
        (isValid, message, stateProfile) = spec.Allows(
            spec.GetOperation<string, ApiResult<{{Name}}Entity>>("Delete"),
            "manual1",
            deleteResponse,
            stateProfile);
        Assert.That(isValid, Is.True, message);

        // Step 4: Get should now return 404
        var getAfterDelete = await GetAsync(_httpClient, "manual1");
        
        (isValid, message, stateProfile) = spec.Allows(
            spec.GetOperation<string, ApiResult<{{Name}}Entity>>("Get"),
            "manual1",
            getAfterDelete,
            stateProfile);
        Assert.That(isValid, Is.True, message);
    }

    private static async Task<ApiResult<{{Name}}Entity>> CreateAsync(HttpClient http, {{Name}}Entity entity)
    {
        var response = await http.PutAsJsonAsync($"/api/entities/{entity.Id}", entity);
        return await ToApiResultAsync(response);
    }

    private static async Task<ApiResult<{{Name}}Entity>> GetAsync(HttpClient http, string id)
    {
        var response = await http.GetAsync($"/api/entities/{id}");
        return await ToApiResultAsync(response);
    }

    private static async Task<ApiResult<{{Name}}Entity>> DeleteAsync(HttpClient http, string id)
    {
        var response = await http.DeleteAsync($"/api/entities/{id}");
        return await ToApiResultAsync(response);
    }

    private static async Task<ApiResult<{{Name}}Entity>> ToApiResultAsync(HttpResponseMessage response)
    {
        var statusCode = (int)response.StatusCode;
        
        if (statusCode >= 200 && statusCode < 300 && response.Content.Headers.ContentLength > 0)
        {
            var data = await response.Content.ReadFromJsonAsync<{{Name}}Entity>();
            return new ApiResult<{{Name}}Entity>(statusCode, data);
        }
        
        return new ApiResult<{{Name}}Entity>(statusCode);
    }
